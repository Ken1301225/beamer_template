% ====================
% Header 模板 —— 章节名 + 页码 + 分段进度条
% 处理无 section 的情况（标题页、目录页）
% ====================
\makeatletter
\setbeamertemplate{headline}{%
  \vskip6pt%
  \leavevmode%
  \hbox to \paperwidth{%
    \hskip0.5cm%
    \ifx\insertsection\@empty%
      \usebeamerfont{section in head}\color{w-muted}%
    \else%
      \usebeamerfont{section in head}\color{w-text}\insertsection%
    \fi%
    \hfill
    {\usebeamerfont{date}\color{w-subtext}\insertframenumber\,/\,\inserttotalframenumber}%
    \hskip0.5cm%
  }%
  \vskip4pt%
  \ifnum\value{totalsections}>0%
    \ifnum\insertsectionnumber>0%
      \leavevmode%
      \hbox to \paperwidth{%
        \hskip0.5cm%
        \foreach \i in {1,...,\value{totalsections}}{%
          \ifnum\i<\insertsectionnumber%
            {\color{w-accent}\rule{\dimexpr(\paperwidth-1cm)/\value{totalsections}-2pt\relax}{4pt}}%
            \hskip2pt%
          \else
            \ifnum\i=\insertsectionnumber%
              {\color{w-rose}\rule{\dimexpr(\paperwidth-1cm)/\value{totalsections}-2pt\relax}{4pt}}%
              \hskip2pt%
            \else
              {\color{w-line}\rule{\dimexpr(\paperwidth-1cm)/\value{totalsections}-2pt\relax}{4pt}}%
              \hskip2pt%
            \fi
          \fi
        }%
        \hfill%
      }%
    \else
      % 没有 section 的页面（标题页、目录页）—— 显示全灰色进度条
      \leavevmode%
      \hbox to \paperwidth{%
        \hskip0.5cm%
        \foreach \i in {1,...,\value{totalsections}}{%
          {\color{w-line}\rule{\dimexpr(\paperwidth-1cm)/\value{totalsections}-2pt\relax}{4pt}}%
          \hskip2pt%
        }%
        \hfill%
      }%
    \fi
  \fi
  \vskip2pt%
  {\color{w-surface}\hrule height 0.6pt}%
  \vskip2pt%
}
\makeatother
